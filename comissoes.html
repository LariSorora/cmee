// Recebe respostas do formulário function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = JSON.parse(e.postData.contents); sheet.appendRow([ data.nome, data.empresa, data.whatsapp, data.email, data.comissao, data.tempo, data.reunioes, data.perfil, data.identifica, data.busca, data.motivacao, data.contribuicao, data.timestamp ]); return ContentService.createTextOutput('OK'); } // Entrega os dados para o dashboard function doGet(e) { const params = e && e.parameter ? e.parameter : {}; const cb = params.callback || ''; const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const linhas = sheet.getDataRange().getValues(); const dados = []; for (var i = 1; i < linhas.length; i++) { var l = linhas[i]; if (!l[0]) continue; dados.push({ nome: l[0], empresa: l[1], whatsapp: l[2], email: l[3], comissao: l[4], tempo: l[5], reunioes: l[6], perfil: l[7], identifica: l[8], busca: l[9], motivacao: l[10], contribuicao: l[11], timestamp: l[12] }); } const json = JSON.stringify(dados); if (cb) { return ContentService .createTextOutput(cb + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
